# install
git clone this repository

# usage
link your config path
```
ln -s /path/dotfiles/fish_prompt.fish ~/.config/fish/functions/fish_prompt.fish                                               
ln -s /path/dotfiles/ide ~/.shellscript/ide                                                                 
ln -s /path/dotfiles/dein_lazy.toml ~/.config/nvim/dein_lazy.toml                                                      
ln -s /path/dotfiles/dein.toml ~/.config/nvim/dein.toml                                                           
ln -s /path/dotfiles/init.vim ~/.config/nvim/init.vim                                                            
ln -s /path/dotfiles/.tmux.conf ~/.tmux.conf                                                                 
ln -s /path/dotfiles/fish_prompt.fish ~/.config/fish/functions/fish_prompt.fish                                              
ln -s /path/dotfiles/.hyper.js ~/.hyper.js                                                                  
ln -s /path/dotfiles/config.fish ~/.config/fish/config.fish
```
